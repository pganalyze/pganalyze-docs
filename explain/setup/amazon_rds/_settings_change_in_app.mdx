export const SettingsChangeNeedsRestart = () => {
  return (
    <>
      <p>Since changes to <code>shared_preload_libraries</code> only take effect at server start, after you apply
      these settings, you will need to restart Postgres:</p>
      <Link className="btn btn-success" to="03_restart_postgres">
        Proceed to Step 3: Restart Postgres
      </Link>
    </>
  )
}

export const SettingsChangeNoRestart = () => {
  return (
    <>
      <p>Since <code>shared_preload_libraries</code> already included <code>auto_explain</code>, after you save the parameter
      group changes, you can skip Step 3 and continue to Step 4:</p>
      <Link className="btn btn-success" to="04_test_and_verify">
        Proceed to Step 4: Test and verify
      </Link>
    </>
  )
}

export const SettingsChangeInApp = ({autoExplainEnabled}) => {
  return autoExplainEnabled ? <SettingsChangeNoRestart /> : <SettingsChangeNeedsRestart /> 
}
